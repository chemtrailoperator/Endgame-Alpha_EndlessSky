mission "Navy Package Delivery"
	name "Package Delivery"
	job
	repeat
	deadline 16
	description "Deliver a package to <destination> by <date> for NavInt. Payment is <payment> credits."
	source
		government "Republic" "Free Worlds"
	destination
		distance 3 6
		government "Republic" "Independent" "Pirate" "Free Worlds"
	to offer
		random < 20
		has "event: EA accepted the endgame"
	on accept
		dialog "The instructions included with the job instruct you to report to the NavInt office. There is a featureless box waiting for you..."
	on complete
		payment 25000
		dialog phrase "EA package delivered"

	name "Package Retrieval"
